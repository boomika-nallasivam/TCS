#include <iostream>
#include <cstdlib>
using namespace std;

int main() {
    int n;
    cin >> n;

    int a[100000];
    for (int i = 0; i < n; i++) cin >> a[i];

    for (int i = 0; i < n; i++) {
        int index = abs(a[i]);

        if (a[index] < 0) {
            cout << index << " ";
        } 
        else {
            a[index] = -a[index];
        }
    }

    return 0;
}
