#include <iostream>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;

    int a[10000];
    for (int i = 0; i < n; i++) cin >> a[i];

    int count = 0;

    for (int start = 0; start < n; start++) {
        int sum = 0;
        for (int end = start; end < n; end++) {
            sum += a[end];
            if (sum == k)
                count++;
        }
    }

    cout << count;
    return 0;
}
