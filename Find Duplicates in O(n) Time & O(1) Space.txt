#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;

    int arr[100];
    for(int i = 0; i < n; i++) {
        cin >> arr[i];
    }

    cout << "Duplicates: ";
    for(int i = 0; i < n; i++) {
        int index = abs(arr[i]) - 1;

        if(arr[index] < 0) {
            cout << index + 1 << " ";
        } else {
            arr[index] = -arr[index];
        }
    }

    return 0;
}
