#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;

    int a[100000];
    for (int i = 0; i < n; i++) cin >> a[i];

    int minPrice = a[0];
    int maxProfit = 0;

    for (int i = 1; i < n; i++) {
        if (a[i] < minPrice)
            minPrice = a[i];

        int profit = a[i] - minPrice;

        if (profit > maxProfit)
            maxProfit = profit;
    }

    cout << maxProfit;
    return 0;
}
