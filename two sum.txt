#include <iostream>
using namespace std;

int main() {
    int n, target;
    cin >> n >> target;

    int a[10000];
    for (int i = 0; i < n; i++) cin >> a[i];

    int index1 = -1, index2 = -1;

    for (int i = 0; i < n; i++) {
        for (int j = i + 1; j < n; j++) {
            if (a[i] + a[j] == target) {
                index1 = i;
                index2 = j;
                break;
            }
        }
        if (index1 != -1) break;
    }

    cout << index1 << " " << index2;
    return 0;
}
